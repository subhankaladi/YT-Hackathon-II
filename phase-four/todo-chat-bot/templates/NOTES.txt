========================================
1. FRONTEND APPLICATION
========================================

Get the frontend URL:
{{- if contains "NodePort" .Values.frontend.service.type }}
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "todo-chat-bot.fullname" . }}-frontend)
  echo "Frontend URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.frontend.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "todo-chat-bot.fullname" . }}-frontend
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "todo-chat-bot.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Frontend URL: http://$SERVICE_IP:{{ .Values.frontend.service.port }}"
{{- else if contains "ClusterIP" .Values.frontend.service.type }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "todo-chat-bot.fullname" . }}-frontend 3000:{{ .Values.frontend.service.port }}
  echo "Frontend URL: http://127.0.0.1:3000"
{{- end }}

========================================
2. BACKEND API
========================================

Get the backend API URL:
{{- if contains "NodePort" .Values.backend.service.type }}
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "todo-chat-bot.fullname" . }}-backend)

  echo "API Documentation: http://$NODE_IP:$NODE_PORT/docs"
{{- else if contains "LoadBalancer" .Values.backend.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "todo-chat-bot.fullname" . }}-backend
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "todo-chat-bot.fullname" . }}-backend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Backend API: http://$SERVICE_IP:{{ .Values.backend.service.port }}"
  echo "API Documentation: http://$SERVICE_IP:{{ .Values.backend.service.port }}/docs"
{{- else if contains "ClusterIP" .Values.backend.service.type }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "todo-chat-bot.fullname" . }}-backend 8000:{{ .Values.backend.service.port }}
  echo "Backend API: http://127.0.0.1:8000"
  echo "API Documentation: http://127.0.0.1:8000/docs"
{{- end }}

========================================
3. MONITOR DEPLOYMENT
========================================

Check pod status:
  kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

Watch deployment:
  kubectl rollout status deployment/{{ include "todo-chat-bot.fullname" . }}-frontend -n {{ .Release.Namespace }}
  kubectl rollout status deployment/{{ include "todo-chat-bot.fullname" . }}-backend -n {{ .Release.Namespace }}

4. VIEW LOGS
========================================

Frontend logs:
  kubectl logs -l app.kubernetes.io/instance={{ .Release.Name }},app=frontend --namespace {{ .Release.Namespace }}

Backend logs:
  kubectl logs -l app.kubernetes.io/instance={{ .Release.Name }},app=backend --namespace {{ .Release.Namespace }}
